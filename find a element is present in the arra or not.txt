        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(); // it is the size of the first array
        int[] arr1 = new int[n];

        // first array input

        for (int i = 0; i < n; i++) {
            arr1[i] = sc.nextInt();
        }
        int x = sc.nextInt(); // element to the searched
       Searched(arr1,n,x);
    }

    static void Searched(int [] arr1,int n, int x){
int result=0;
        for(int i=0; i<n; i++){
            if(arr1[i]==x){
             result=x;
                System.out.println("element is present at the index " + i);
            }
        }

        if(result==0){
            System.out.println("element is not present in the array");
        }


    }
}
