import java.util.*;

public class NEW {

    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt(); // it is the size of the first array
        int m=sc.nextInt(); // it is the size of the second array
        int []arr1=new int[n];
        int [] arr2=new int[m];
        int result=0;

        // first array input

        for(int i=0;  i<n; i++){
            arr1[i]=sc.nextInt();
        }

        //input of the second array
        for(int i=0; i<m; i++){
            arr2[i]=sc.nextInt();
        }
        // here we wil do sum of the first array
        for(int i=0; i<n; i++){
            result+=arr1[i];
        }
        // here we wil some of teh second array

        for(int i=0; i<m; i++){
            result+=arr2[i];
        }

        System.out.println(result);
    }
}
